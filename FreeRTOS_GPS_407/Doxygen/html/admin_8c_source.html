264void CreateTasks(void)
265{
266 PTASKDATA ptd = tasks; // tasks == &tasks[0]: both are addresses of first struct
267 vTaskSuspendAll(); // zet de scheduler stil en dwing FreeRTOS om eerst alle taken te starten
268
269 for (; ptd->func != NULL; ptd++) // loopje zonder teller; gewoon de pointer steeds ophogen...
270 {
271 UART_puts("\r\nCreated: ");
272 UART_puts((char *)ptd->attr.name);
273
274 // nb. kijk naar de haakjes van argument 3: het gaat om het adres van (structure attr).
275 ptd->hTask = osThreadNew(ptd->func, ptd->argument, &(ptd->attr));
276 }
277
278 xTaskResumeAll(); // start nu de scheduler: play ball
279 CheckTaskData(TRUE); // display alle taskdata op UART with output to terminal
280}

339void StartStopTask(int task_id)
340{
341 PTASKDATA ptd = tasks;
342 unsigned int task_nr;
343 TaskStatus_t xTaskDetails;
344
345 for (task_nr = 1; ptd->func != NULL; ptd++, task_nr++)
346 if (task_nr == task_id) // hebbes
347 {
348 vTaskGetInfo (ptd->hTask, &xTaskDetails, pdTRUE, eInvalid);
349 UART_puts("\r\n");
350 UART_puts(ptd->attr.name);
351
352 // een task kan meerdere statussen hebben; het wordt simpeler
353 // als je alleen de eSuspended checkt.
354 if (xTaskDetails.eCurrentState == eSuspended)
355 {
356 vTaskResume(ptd->hTask);
357 UART_puts(" resumed.");
358 }
359 else // bv. eBlocked, eRunning, eInvalid etc.
360 {
361 vTaskSuspend(ptd->hTask);
362 UART_puts(" suspended.");
363 }
364 return;
365 }

373void CheckTaskData(int output)
374{
375 PTASKDATA ptd = tasks;
376 UBaseType_t highwatermark;
377 TaskStatus_t xTaskDetails;
378 unsigned int task_nr;
379 unsigned long totalalloc = 0;
380 unsigned long stacksize, free, used;
381 int i;
382
383 for (task_nr = 1; ptd->func != NULL; ptd++, task_nr++)
384 {
385 highwatermark = uxTaskGetStackHighWaterMark(ptd->hTask); // amount of free bytes
386 stacksize = ptd->attr.stack_size * 4;
387 free = highwatermark * 4;
388 used = 100 - ((free * 100) / stacksize); // in percentage
389 totalalloc += stacksize;
390
391 vTaskGetInfo (ptd->hTask, &xTaskDetails, pdTRUE, eInvalid); // get task status
392 if (output)
393 {
394 UART_puts("\r\n\t["); UART_putint(task_nr); UART_puts("] ");
395 UART_puts(ptd->attr.name);
396 UART_puts("\t priority: "); UART_putint(ptd->attr.priority);
397 UART_puts("\t stacksize: "); UART_putint(stacksize); UART_puts(" bytes");
398 UART_puts("\t free: "); UART_putint(free); UART_puts(" bytes");
399 UART_puts("\t used: "); UART_putint(used); UART_puts("%");
400 UART_puts("\t status: "); UART_puts(xTaskDetails.eCurrentState == eSuspended ? "suspended": "running");
401 }
402
403 if (free < (10 * 4)) // oops, too little stackspace reserved
404 {
405 BUZZER_put(1000);
406 Uart_debug_out = DEBUG_OUT_NONE;
407 osDelay(750);
408 UART_puts("\r\n==== ERROR NO STACKSPACE LEFT =====\r\n");
409 UART_puts("\r\n\t["); UART_putint(task_nr); UART_puts("] ");
410 UART_puts(ptd->attr.name);
411 UART_puts("\t stacksize: "); UART_putint(stacksize); UART_puts(" bytes");
412 UART_puts("\t free: "); UART_putint(free); UART_puts(" bytes");
413 UART_puts("\t used: "); UART_putint(used); UART_puts("%\r\n");
414
415 for (i=0; i<10; i++) // extra visual output
416 {
417 BUZZER_put(1000);
418 LED_put(0); // set leds
419 osDelay(750);
420 LED_put(255); // set leds
421 }
422 error_HaltOS("Error Task Stackspace");
423 }
424 }
425
426 if (output)
427 {
428 UART_puts("\r\n\tTotal heap: "); UART_putint(configTOTAL_HEAP_SIZE);
429 UART_puts(" Allocated task stack: "); UART_putint(totalalloc);
430 UART_puts(" Free heap space: "); UART_putint(configTOTAL_HEAP_SIZE - totalalloc);
431 UART_puts("\r\n");
432 }
433}
