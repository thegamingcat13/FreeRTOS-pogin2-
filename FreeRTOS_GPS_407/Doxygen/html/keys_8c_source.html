33void KEYS_init(void)
34/* Keys initialize
35 * In this function the ports are set for the keyboard.
36 * The rows are outputs and the columns are input.
37 */
38{
39 __HAL_RCC_GPIOD_CLK_ENABLE();
40 GPIO_InitTypeDef gpio;
41 gpio.Mode = GPIO_MODE_OUTPUT_PP;
42 gpio.Pull = GPIO_NOPULL;
43 gpio.Speed = GPIO_SPEED_FREQ_VERY_HIGH;
44
45 gpio.Pin = KEY_ROW;
46 HAL_GPIO_Init(PKEY_ROW, &gpio);
47
48 gpio.Pin = KEY_COL;
49 gpio.Mode = GPIO_MODE_INPUT;
50 gpio.Pull = GPIO_PULLDOWN;
51 HAL_GPIO_Init(PKEY_COL, &gpio);
52 HAL_GPIO_WritePin(PKEY_ROW, KEY_ROW, 0);
53}
54
55void KEYS_initISR(int hilo)
56/* Keys initialize for ISR
57 * Set all lines high
58 * */
59{
60 HAL_GPIO_WritePin(PKEY_ROW, KEY_R1, hilo);
61 HAL_GPIO_WritePin(PKEY_ROW, KEY_R2, hilo);
62 HAL_GPIO_WritePin(PKEY_ROW, KEY_R4, hilo);
63 HAL_GPIO_WritePin(PKEY_ROW, KEY_R3, hilo);
64}

67unsigned int KEYS_read(void)
68/* Keys read
69 * This function reads which key is pressed.
70 * It does so by making a row high and check if there is a connection with a column.
71 * If there is a connection the value is returned.
72 */
73{
74 unsigned int key = 0;
75 unsigned out = 0;
76
77 HAL_GPIO_WritePin(PKEY_ROW, KEY_R1, 0);
78 HAL_GPIO_WritePin(PKEY_ROW, KEY_R4, 1);
79 key = KEYS_kolom();
80 if( key != 0 ) out= 12+key;
81
82 HAL_GPIO_WritePin(PKEY_ROW, KEY_R4, 0);
83 HAL_GPIO_WritePin(PKEY_ROW, KEY_R3, 1);
84 key = KEYS_kolom();
85 if( key != 0 ) out = 8+key;
86
87 HAL_GPIO_WritePin(PKEY_ROW, KEY_R3, 0);
88 HAL_GPIO_WritePin(PKEY_ROW, KEY_R2, 1);
89 key = KEYS_kolom();
90 if( key != 0 ) out = 4+key;
91
92 HAL_GPIO_WritePin(PKEY_ROW, KEY_R2, 0);
93 HAL_GPIO_WritePin(PKEY_ROW, KEY_R1, 1);
94 key = KEYS_kolom();
95 if( key != 0 ) out = key;
96
97 return out;
98}
99
100unsigned int KEYS_kolom(void)
101/* Keys Column
102 * In this function the columns are checked if there is a connection.
103 * If a column has a connection the return value is the column number.
104 */
105{
106 unsigned int key = 0;
107 if (HAL_GPIO_ReadPin(PKEY_COL, KEY_K1) == 1) key = 1;
108 if (HAL_GPIO_ReadPin(PKEY_COL, KEY_K2) == 1) key = 2;
109 if (HAL_GPIO_ReadPin(PKEY_COL, KEY_K3) == 1) key = 3;
110 if (HAL_GPIO_ReadPin(PKEY_COL, KEY_K4) == 1) key = 4;
111 return key;
112}
