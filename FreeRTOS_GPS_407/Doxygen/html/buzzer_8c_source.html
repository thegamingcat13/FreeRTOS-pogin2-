24void BUZZER_set(int counter)
25{
26 while(counter--)
27 {
28 HAL_GPIO_WritePin(Buzzer_GPIO_Port, Buzzer_Pin, GPIO_PIN_SET);
29 HAL_Delay(1);
30 HAL_GPIO_WritePin(Buzzer_GPIO_Port, Buzzer_Pin, GPIO_PIN_RESET);
31 HAL_Delay(1);
32 }
33}
34
35void Buzzer_put(unsigned int counter)
36{
37 BUZZER_set(counter);
38 HAL_Delay(150);
39}
40
41// Maak een piepje met meegegeven lengte
42// toevoeging om compatibel te blijven met ARM v4.2
43void BUZZER_put(unsigned int time)
44{
45 volatile unsigned int i;
46 volatile unsigned int k=0;
47
48 while (k++<100)
49 {
50 HAL_GPIO_WritePin(Buzzer_GPIO_Port, Buzzer_Pin, GPIO_PIN_SET);
51 for (i=0;i<(time+k);i++);
52 HAL_GPIO_WritePin(Buzzer_GPIO_Port, Buzzer_Pin, GPIO_PIN_RESET);
53 for (i=0;(i<time+k);i++);
54 }
55}
