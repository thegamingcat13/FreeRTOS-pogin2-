91void CreateTasks(void)

94 vTaskSuspendAll(); // zet de scheduler stil en dwing FreeRTOS om eerst alle taken te starten

166void StartStopTask(int task_id)
167{
168 PTASKDATA ptd = tasks;
169 unsigned int task_nr;
170 TaskStatus_t xTaskDetails;
171
172 for (task_nr = 1; ptd->func != NULL; ptd++, task_nr++)
173 if (task_nr == task_id) // hebbes
174 {
175 vTaskGetInfo (ptd->hTask, &xTaskDetails, pdTRUE, eInvalid);
176 UART_puts("\r\n");
177 UART_puts(ptd->attr.name);
178
179 // een task kan meerdere statussen hebben; het wordt simpeler
180 // als je alleen de eSuspended checkt.
181 if (xTaskDetails.eCurrentState == eSuspended)
182 {
183 vTaskResume(ptd->hTask);
184 UART_puts(" resumed.");
185 }
186 else // bv. eBlocked, eRunning, eInvalid etc.
187 {
188 vTaskSuspend(ptd->hTask);
189 UART_puts(" suspended.");
190 }
191 return;
192 }
193}

200void DisplayTaskData(void)
201{
202 PTASKDATA ptd = tasks;
203 UBaseType_t freespace;
204 TaskStatus_t xTaskDetails;
205 unsigned int used;
206 unsigned int task_nr; // tasknr for changing priority
207
208 for (task_nr=1; ptd->func != NULL; ptd++, task_nr++)
209 {
210 freespace = uxTaskGetStackHighWaterMark(ptd->hTask) * 4; // amount of free bytes
211 used = 100 - (freespace * 100) / ptd->attr.stack_size; // in percentage
212
213 vTaskGetInfo (ptd->hTask, &xTaskDetails, pdTRUE, eInvalid); // get task status
214
215 UART_puts("\r\n\t["); UART_putint(task_nr); UART_puts("] ");
216 UART_puts(ptd->attr.name);
217 UART_puts("\t priority: "); UART_putint(ptd->attr.priority);
218 UART_puts("\t stacksize: "); UART_putint(ptd->attr.stack_size);
219 UART_puts("\t used: "); UART_putint(used); UART_puts("%");
220 UART_puts("\t status: "); UART_puts(xTaskDetails.eCurrentState == eSuspended ? "suspended": "running");
221 }
222 UART_puts("\r\n");
223}
