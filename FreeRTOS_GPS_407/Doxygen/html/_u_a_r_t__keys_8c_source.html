61void UART_keys_IRQ (void *argument)
62{
63 char buffer[QSIZE_UART];
64 char buffer_copy[QSIZE_UART];
65 int pos = 0;
66 int finish = FALSE;
67 osThreadId_t hTask;
68
69 UART_puts((char *)__func__); UART_puts("started, enter key on terminal...\r\n");
70
71 if (!(hTask = xTaskGetHandle("UART_menu")))
72 error_HaltOS("Err:UART_hndl 0");
73
74 while(TRUE)
75 {
76 // laat de VIC de q vullen, die per char in onze eigen string geplaatst wordt
77 // tot LFEED gedrukt is of qsize-1 bereikt is.
78 // nb: argument 2 is het adres waar ik het q-char in zet
79 // nb: q-receive haalt gelijk de buffer leeg (q-peek niet).
80 xQueueReceive(hUART_Queue, &buffer[pos], portMAX_DELAY);
81
82 // negeer dit char bij geen data: -1, 255, of CR of spatie
83 if (buffer[pos] == 0 || buffer[pos] == -1 || buffer[pos] == 255 || buffer[pos] == CRETURN || buffer[pos] == 32)
84 continue;
85
86 //UART_putchar(buffer[pos]); // echo
87
88 // check of de string gesloten was of gesloten moet worden...
89 if (buffer[pos] == LFEED) // close als LF gedrukt
90 {
91 buffer[pos] = '\0';
92 finish = TRUE;
93
94 }
95 else if (pos == QSIZE_UART - 2) // close if end of buf
96 {
97 buffer[++pos] = '\0'; // first, skip to last position, then close string
98 finish = TRUE;
99 }
100
101 if (finish)
102 {
103 // de volgende taak krijgt een copy van de string
104 // mijn eigen buffer kan zo gelijk weer gevuld worden door de ISR
105 strcpy(buffer_copy, buffer);
106 memset(buffer, 0, QSIZE_UART); // clear original buffer
107 finish = FALSE;
108 pos = 0;
109
110 xTaskNotify(hTask, buffer_copy, eSetValueWithOverwrite); // notify task2 with copy
111
112 if (Uart_debug_out & UART_DEBUG_OUT)
113 {
114 UART_puts("\n\r"); UART_puts((char *)__func__);
115 UART_puts(": UART_menu notified with "); UART_puts(buffer_copy);
116 UART_puts("\r\n");
117 }
118 }
119 else
120 pos++;
121 }
122}

142 // wacht op de string; let op de cast van (unsigned long) naar (char *)!

146 // check het eerste (en vaak enige) karakter van de inkomende command-string
